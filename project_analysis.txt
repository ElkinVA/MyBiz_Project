def load_more_products(request):

    if not is_ajax:

    # –°–ø–æ—Å–æ–± 3: –î–ª—è –æ—Ç–ª–∞–¥–∫–∏ –∏ —Ç–µ—Å—Ç–æ–≤ - –ø–∞—Ä–∞–º–µ—Ç—Ä ?debug=ajax

    if is_ajax or debug_mode:
        page = request.GET.get('page', 2)  # –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é –∑–∞–≥—Ä—É–∂–∞–µ–º –≤—Ç–æ—Ä—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É
        category_slug = request.GET.get('category', None)

        products = Product.objects.filter(is_active=True)

        if category_slug and category_slug != 'all':
            try:
                category = Category.objects.get(slug=category_slug, is_active=True)
                products = products.filter(category=category)
            except Category.DoesNotExist:
                category = None

        paginator = Paginator(products, 12)

        try:
            products_page = paginator.page(page)
        except (EmptyPage, PageNotAnInteger):
            return JsonResponse({'html': '', 'has_next': False, 'next_page': None})

        # –†–µ–Ω–¥–µ—Ä–∏–º —Ç–æ–ª—å–∫–æ HTML —Å —Ç–æ–≤–∞—Ä–∞–º–∏
        html = render_to_string('products/product_items.html', {'products': products_page})

        return JsonResponse({
            'html': html,
            'has_next': products_page.has_next(),
            'next_page': products_page.next_page_number() if products_page.has_next() else None
        })

    # –ï—Å–ª–∏ –∑–∞–ø—Ä–æ—Å –Ω–µ AJAX, –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –æ—à–∏–±–∫—É 400
    return JsonResponse({
        'error': 'This endpoint requires an AJAX request',
        'detail': 'Set X-Requested-With header to XMLHttpRequest'
    }, status=400)

from .models import SiteSettings, Promotion
from pages.models import Page

def site_settings(request):
    """–î–æ–±–∞–≤–ª—è–µ—Ç –Ω–∞—Å—Ç—Ä–æ–π–∫–∏ —Å–∞–π—Ç–∞ –≤ –∫–æ–Ω—Ç–µ–∫—Å—Ç –≤—Å–µ—Ö —à–∞–±–ª–æ–Ω–æ–≤"""
    settings = SiteSettings.load()
    return {'site_settings': settings}

def promotions(request):
    """–î–æ–±–∞–≤–ª—è–µ—Ç –∞–∫—Ç–∏–≤–Ω—ã–µ –ø—Ä–æ–º–æ-–∞–∫—Ü–∏–∏ –≤ –∫–æ–Ω—Ç–µ–∫—Å—Ç"""
    active_promotions = Promotion.objects.filter(is_active=True).order_by('-created_at')[:3]
    return {'promotions': active_promotions}

def header_pages(request):
    """–î–æ–±–∞–≤–ª—è–µ—Ç —Å—Ç—Ä–∞–Ω–∏—Ü—ã –¥–ª—è —à–∞–ø–∫–∏ —Å–∞–π—Ç–∞"""
    pages = Page.get_header_pages()
    return {'header_pages': pages}

    Alpine.data('productLoader', () => ({
        loading: false,
        nextPage: {% if products.has_next %}{{ products.next_page_number }}{% else %}null{% endif %},
        hasNext: {{ products.has_next|yesno:"true,false" }},
        categorySlug: '',
        loadedCount: {{ products|length }}, // –ù–∞—á–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–≥—Ä—É–∂–µ–Ω–Ω—ã—Ö —Ç–æ–≤–∞—Ä–æ–≤

        init(categorySlug) {
            this.categorySlug = categorySlug;
        },

        async loadMore() {
            if (this.loading || !this.hasNext || !this.nextPage) return;

            this.loading = true;

            try {
                const url = `/products/load-more/?page=${this.nextPage}&category=${this.categorySlug}`;
                const response = await fetch(url, {
                    headers: {
                        'X-Requested-With': 'XMLHttpRequest'
                    }
                });

                if (!response.ok) throw new Error('–û—à–∏–±–∫–∞ —Å–µ—Ç–∏');

                const data = await response.json();

                if (data.html) {

                    const newItems = document.createElement('div');

def transliterate_ru_to_en(text):
    """–¢—Ä–∞–Ω—Å–ª–∏—Ç–µ—Ä–∞—Ü–∏—è —Ä—É—Å—Å–∫–∏—Ö –±—É–∫–≤ –≤ –ª–∞—Ç–∏–Ω—Å–∫–∏–µ"""
    conversion = {
        '–∞': 'a', '–±': 'b', '–≤': 'v', '–≥': 'g', '–¥': 'd', '–µ': 'e', '—ë': 'yo',
        '–∂': 'zh', '–∑': 'z', '–∏': 'i', '–π': 'y', '–∫': 'k', '–ª': 'l', '–º': 'm',
        '–Ω': 'n', '–æ': 'o', '–ø': 'p', '—Ä': 'r', '—Å': 's', '—Ç': 't', '—É': 'u',
        '—Ñ': 'f', '—Ö': 'h', '—Ü': 'ts', '—á': 'ch', '—à': 'sh', '—â': 'sch',
        '—ä': '', '—ã': 'y', '—å': '', '—ç': 'e', '—é': 'yu', '—è': 'ya',
        '–ê': 'A', '–ë': 'B', '–í': 'V', '–ì': 'G', '–î': 'D', '–ï': 'E', '–Å': 'YO',
        '–ñ': 'Zh', '–ó': 'Z', '–ò': 'I', '–ô': 'Y', '–ö': 'K', '–õ': 'L', '–ú': 'M',
        '–ù': 'N', '–û': 'O', '–ü': 'P', '–†': 'R', '–°': 'S', '–¢': 'T', '–£': 'U',
        '–§': 'F', '–•': 'H', '–¶': 'Ts', '–ß': 'Ch', '–®': 'Sh', '–©': 'Sch',
        '–™': '', '–´': 'Y', '–¨': '', '–≠': 'E', '–Æ': 'Yu', '–Ø': 'Ya'
    }

    result = []
    for char in text:
        if char in conversion:
            result.append(conversion[char])
        elif char.isalnum() or char in ['-', '_']:
            result.append(char)
        else:
            result.append('-')

    slug = ''.join(result)
    # –£–¥–∞–ª—è–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã–µ –¥–µ—Ñ–∏—Å—ã
    slug = re.sub(r'-+', '-', slug)
    # –£–¥–∞–ª—è–µ–º –¥–µ—Ñ–∏—Å—ã –≤ –Ω–∞—á–∞–ª–µ –∏ –∫–æ–Ω—Ü–µ
    slug = slug.strip('-')

    return slug.lower()
